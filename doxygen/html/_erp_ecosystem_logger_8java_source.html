195  /*
196  I believe that all kind of log method requires level to decide
197  what kind of file need to be generated and what format to use while
198  recording the log. So this decision can be made centrally. That is why I
199  designed this central method to produce a logger object with the decision of
200  a) The level of the log
201  b) which class is generating the log
202 
203  That is why this method requies two parameters
204  a) errorClass : where the exception is generated
205  b) level of the log.
206  */
207  private org.apache.log4j.Logger log(Class errorClass,Level level)
208  {
209  /*
210  If the user of the system do not want to supply error generating class
211  then by default the current class will be considered to be the class which is
212  generating the eeror.
213  */
214  org.apache.log4j.Logger logger=null;
215  if(errorClass == null)
216  logger= org.apache.log4j.Logger.getLogger(this.getClass().getName());
217  else
218  /*
219 
220  */
221  logger=org.apache.log4j.Logger.getLogger(errorClass.getName());
222  /*
223  If the user want to record the log into respective file then the rollingOn
224  should be true.
225  */
226  if(rollingOn)
227  {
228  try {
229  /*
230  I want to record different level of log into different files.
231  So what file wiil be used will depened upon the level.
232  */
233  String fileName="default.log";
234  if(level == Level.ALL)
235  fileName = all_log_file_name;
236  else if(level == Level.DEBUG)
237  fileName = debug_log_file_name;
238  else if(level == Level.ERROR)
239  fileName = error_log_file_name;
240  else if(level == Level.FATAL)
241  fileName = fatal_log_file_name;
242  else if(level == Level.INFO)
243  fileName = info_log_file_name;
244  else if(level == Level.WARN)
245  fileName = warn_log_file_name;
246 
247  logger.addAppender(getRollingFileAdapter(fileName));
248  } catch (URISyntaxException e) {
249  java.util.logging.Logger.getLogger(this.getClass().getName()).log(java.util.logging.Level.SEVERE,
250  "URL Is Not Valid For Log File\nUnable To Create Daily LOG FILE");
251  logger.addAppender(new ConsoleAppender());
252  }
253  }
254  logger.setLevel(level);
255  return logger;
256  }
263  private DailyRollingFileAppender getRollingFileAdapter(String fileName) throws URISyntaxException
264  {
265  URI logDIRFullName = this.getClass().getResource("/").toURI();
266  String normalizePath = Paths.get(logDIRFullName).toString();
267 
268  String log_file_full_name = normalizePath + "/" + log_directory_name + "/" + fileName;
269  PatternLayout layout = new PatternLayout();
270  String conversionPattern = getConversionPattern();
271  layout.setConversionPattern(conversionPattern);
272  DailyRollingFileAppender rollingAppender = new DailyRollingFileAppender();
273  rollingAppender.setFile(log_file_full_name);
274  rollingAppender.setDatePattern("'.'"+datePattern);
275  rollingAppender.setLayout(layout);
276  rollingAppender.activateOptions();
277  return rollingAppender;
278  }
